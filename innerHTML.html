Ventana Modal

Este es un ejemplo de ventana modal simple creada con HTML, CSS y JavaScript.

Puedes cerrarla haciendo clic en la 'X' o fuera de este cuadro.

function renderList() { dynamicList.innerHTML = ''; // Limpiar lista actual items.forEach((itemText, index) => { const li = document.createElement('li'); li.textContent = itemText; const deleteBtn = document.createElement('button'); deleteBtn.textContent = 'X'; deleteBtn.onclick = () => { deleteItem(index); }; li.appendChild(deleteBtn); dynamicList.appendChild(li); }); } function addItem() { const newItemText = itemInput.value.trim(); if (newItemText) { items.push(newItemText); // Añadir al array itemInput.value = ''; // Limpiar input saveAndRender(); } else { alert('Por favor, introduce texto para el elemento.'); } } function deleteItem(index) { items.splice(index, 1); // Eliminar del array saveAndRender(); } function saveAndRender() { localStorage.setItem('dynamicListItems', JSON.stringify(items)); // Guardar en Local Storage renderList(); // Volver a dibujar la lista } addItemBtn.addEventListener('click', addItem); itemInput.addEventListener('keypress', (event) => { // Permitir añadir con la tecla Enter if (event.key === 'Enter') { addItem(); } }); // Renderizar la lista inicial al cargar la página renderList();
